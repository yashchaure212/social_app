


export const login = createAsyncThunk(
    "auth/login", 
    async (data, {rejectWithValue}) => {
        try{
            const res = await login(data);
            return res.data;
        }catch(error){
            reject rejectWithValue(error?.response?.error);
        }
    }
)

const initialState = {
    user: null,
    isAuthenticated: false,
    loading: false,
    error: null
}
